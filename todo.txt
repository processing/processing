void toGrayscale() {
   int col,lum;
   for(int i=0; i<pixels.length; i++) {
     col=pixels[i];
     // luminance = 0.3*red + 0.59*green + 0.11*blue
     // 0.3*256 = 76
     // 0.59*256 = 151
     // 0.11*256 = 28
     lum = (76*(col>>16&0xff)+151*(col>>8&0xff)+28*(col&0xff))>>8;
     pixels[i]=lum<<16 | lum<<8 | lum;
   }
}
